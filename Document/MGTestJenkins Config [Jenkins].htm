#!/bin/bash -l
export LANG=en_US.UTF-8
export LANGUAGE=en_US.UTF-8
export LC_ALL=en_US.UTF-8
 
# shell脚本打包需要手动配置证书
 
# 证书,这里使用生产证书与上面打包环境进行对应
CODE_SIGN_DISTRIBUTION="Apple Distribution: National Center For Open & Distance Education Limited Company (SAD9GVSU92)"
 
# 工程名
APP_NAME="MGTestJenkins"
# 蒲公英安装地址
APP_DOWNPATH="https://www.pgyer.com/MGOnline"
 
#   输出属性列表
Scheme="MGTestJenkins"
ExportOptionsPList="ExportOptions.plist"
PList="Info.plist"
CONFIGURATION="release"
 
 
# 测试环境
if [ ${environment} = "MGTestJenkinsBeta" ];then
    APP_NAME="MGTestJenkinsBeta"
    APP_DOWNPATH="https://www.pgyer.com/MGBeta"
    Scheme="MGTestJenkinsBeta"
    ExportOptionsPList="ExportOptions_beta.plist"
    PList="BetaInfo.plist"
fi
 
# 开发环境
if [ ${environment} = "MGTestJenkinsDev" ];then
    APP_NAME="MGTestJenkinsDev"
    APP_DOWNPATH="https://www.pgyer.com/MGDev"
    Scheme="MGTestJenkinsDev"
    ExportOptionsPList="ExportOptions_dev.plist"
    PList="DevInfo.plist"
fi
 
echo "👉 plist内容：${ExportOptionsPList}"
 
 
# info.plist路径
project_infoplist_path="${WORKSPACE}/${APP_NAME}/${PList}"
 
echo "👉 info.plist路径：${project_infoplist_path}"
 
# 取版本号
bundleShortVersion=$(/usr/libexec/PlistBuddy -c "print CFBundleShortVersionString" "${project_infoplist_path}")
# 取build值
bundleVersion=$(/usr/libexec/PlistBuddy -c "print CFBundleVersion" "${project_infoplist_path}")
 
# 要上传的ipa文件路径,这个文件路径需要确定,打包输出的时候,输出到这个指定位置,以后从这个位置上传.ipa文件
IPA_PATH="$WORKSPACE/build/${BUILD_TAG}"
 
# 打印指定的文件位置
echo "👉  IPA_PATH:${IPA_PATH}"
 
echo "=================clean================="
xcodebuild clean  -workspace  ${APP_NAME}.xcworkspace \
                  -scheme ${Scheme} \
                  -configuration ${CONFIGURATION}
                  
                  
echo "================= clean over ================="
 
echo "=================开始生成 .xcarchive================="
# 构建ipa包
xcodebuild archive  -workspace ${APP_NAME}.xcworkspace \
                    -scheme ${Scheme} \
                    -configuration ${Configuration} \
                    CFBundleVersion=${bundleVersion} \
                    -destination generic/platform=ios \
                    -archivePath build/${BUILD_TAG}.xcarchive \
                    CODE_SIGN_IDENTITY="${CODE_SIGN_DISTRIBUTION}"
 
echo "=================生成 .xcarchive 完成================="
 
 
#1 -workspace：项目名称.xcworkspace（或-project：项目名称.xcodeproj）              
#2 -scheme：通常默认为项目名称              
#3 -configuration：配置（Release，或Debug；未设置时默认为Release）              
#4 -archivePath：archive包保存目录              
#5 CODE_SIGN_IDENTITY：证书（证书名称）              
#6 PROVISIONING_PROFILE：描述文件UUID
 
 
echo "=================开始导出.ipa================="
xcodebuild -exportArchive -archivePath build/${BUILD_TAG}.xcarchive \
                          -exportPath ${IPA_PATH} \
                          -destination generic/platform=ios \
                          -exportOptionsPlist ${ExportOptionsPList} \
                          -allowProvisioningUpdates
 
echo "👉 ipa包构建完成啦~~ 🚀🚀🚀🚀"
echo "👉 ipa导出路径：IPA_PATH:${IPA_PATH}"
